417 void cmopad_eulerToMatrix(double alpha, double beta, double gamma,
418  double mat[3][3])
419 {
420  double ca, cb, cg, sa, sb, sg;
421  ca = cos(alpha);
422  cb = cos(beta);
423  cg = cos(gamma);
424  sa = sin(alpha);
425  sb = sin(beta);
426  sg = sin(gamma);
427 
428  mat[0][0] = cb*cg-ca*sb*sg; mat[0][1] = sb*cg+ca*cb*sg; mat[0][2] = sa*sg;
429  mat[1][0] =-cb*sg-ca*sb*cg; mat[1][1] =-sb*sg+ca*cb*cg; mat[1][2] = sa*cg;
430  mat[2][0] = sa*sb; mat[2][1] =-sa*cb; mat[2][2] = ca;
431  return;
432 }

686 void cmopad_MatrixToEuler(double rotmat[3][3],
687  double *alpha, double *beta, double *gamma)
688 {
689  const int n3 = 3;
690  int inc = 1;
691  double ex[3], ez[3], exs[3], ezs[3], enodes[3], enodess[3],
692  xnorm, cos_alpha, a,b,g,a1,a2;
693  double twopi = 2.0*M_PI;
694  int i,j;
695  //------------------------------------------------------------------------//
696  //
697  // transpose matrix vector multiply
698  ex[0] = 1.0; ex[1] = 0.0; ex[2] = 0.0;
699  ez[0] = 0.0; ez[1] = 0.0; ez[2] = 1.0;
700  for (i=0; i<3; i++)
701  {
702  exs[i] = 0.0; ezs[i] = 0.0;
703  for (j=0; j<3; j++)
704  {
705  exs[i] = exs[i] + rotmat[j][i]*ex[j];
706  ezs[i] = ezs[i] + rotmat[j][i]*ez[j];
707  }
708  }
709  // Cross product
710  array_cross3(ez, ezs, enodes);
711  xnorm = sqrt(pow(enodes[0],2) + pow(enodes[1],2) + pow(enodes[2],2));
712  if (xnorm < 1.e-10)
713  {
714  for (i=0; i<3; i++)
715  {
716  enodes[i] = exs[i];
717  }
718  }
719  for (i=0; i<3; i++)
720  {
721  enodess[i] = 0.0;
722  for (j=0; j<3; j++)
723  {
724  enodess[i] = enodess[i] + rotmat[i][j]*enodes[j];
725  }
726  }
727  // Multiply
728  cos_alpha = cblas_ddot(n3, ez, inc, ezs, inc);
729  cos_alpha = fmin( 1.0, cos_alpha);
730  cos_alpha = fmax(-1.0, cos_alpha);
731  a = acos(cos_alpha);
732  a1 = atan2( enodes[1], enodes[0]);
733  a2 =-atan2(enodess[1], enodess[0]);
734  b = numpy_mod(a1, twopi);
735  g = numpy_mod(a2, twopi);
736  // Compute unique Euler angles
737  cmopad_uniqueEuler(&a, &b, &g);
738  *alpha = a;
739  *beta = b;
740  *gamma = g;
741  return;
